        let currentLang = 'ar';

        function renderSpecializations() {
            const specializationsList = document.getElementById('specializations-list');
            specializationsList.innerHTML = '';
            translations[currentLang].specializations.forEach(spec => {
                const item = document.createElement('div');
                item.className = 'specialization-item';
                item.innerHTML = `<h3><a href="#" onclick="showSection('dynamic-content'); showContent('${spec.id}'); return false;">${spec.name}</a></h3>`;
                specializationsList.appendChild(item);
            });
        }

        function showSection(sectionId) {
            document.querySelectorAll('main section').forEach(section => {
                section.style.display = 'none';
            });
            document.getElementById(sectionId).style.display = 'block';
            // Show specializations section after hero
            if (sectionId === 'home-section') {
                document.getElementById('specializations-section').style.display = 'block';
            }
        }
        
        function showContent(contentId) {
            const content = translations[currentLang].content[contentId];
            document.getElementById('content-container').innerHTML = content;
            showSection('dynamic-content');
            return false;
        }
        
        function switchLanguage(lang) {
            currentLang = lang;
            const langData = translations[lang];

            document.documentElement.lang = lang;
            document.documentElement.dir = langData.direction;
            
            document.title = langData.title;
            document.getElementById('site-title').textContent = langData.siteTitle;
            document.getElementById('hero-title').textContent = langData.heroTitle;
            document.getElementById('hero-subtitle').textContent = langData.heroSubtitle;
            document.getElementById('specializations-title').textContent = langData.specializationsTitle;
            document.getElementById('contact-title').textContent = langData.contactTitle;
            document.getElementById('contact-address').textContent = langData.contactAddress;
            document.getElementById('contact-phone').innerHTML = langData.contactPhone;
            document.getElementById('contact-email').textContent = langData.contactEmail;
            document.getElementById('whatsapp-link').textContent = langData.whatsappLink;
            document.getElementById('whatsapp-link').href = `https://wa.me/96171856207`;
            document.getElementById('footer-text').innerHTML = langData.footerText;

            renderSpecializations();
            showSection('home-section');
            document.getElementById('specializations-section').style.display = 'block'; // Ensure specializations show after home
            document.getElementById('contact-section').style.display = 'block';
            document.getElementById('dynamic-content').style.display = 'none';


            // Set active link
            document.querySelectorAll('.lang-switcher a').forEach(link => {
                link.classList.remove('active');
            });
            document.querySelector(`.lang-switcher a[onclick*="'${lang}'"]`).classList.add('active');
        }

        // Initial setup on page load
        window.onload = function() {
            switchLanguage('ar');
        };
